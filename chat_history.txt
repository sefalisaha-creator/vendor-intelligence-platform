1. backend/src/controllers/vendorController.js
JavaScript
// Path: backend/src/controllers/vendorController.js
const Vendor = require('../models/Vendor');
const Ledger = require('../models/Ledger');

// Fetch all vendors with dynamic filtering
exports.getVendors = async (req, res, next) => {
  try {
    const { city, category, status } = req.query;
    let queryFilter = {};

    // Build the query object dynamically based on present search parameters
    if (city) queryFilter.city = { $regex: new RegExp(city, 'i') }; // Case-insensitive exact/partial matches
    if (category) queryFilter.category = { $regex: new RegExp(category, 'i') };
    if (status) queryFilter.status = status; // Exact matching for strict enums

    const vendors = await Vendor.find(queryFilter);
    res.status(200).json({ success: true, count: vendors.length, data: vendors });
  } catch (error) {
    next(error);
  }
};

// Fetch a single vendor and populate their corresponding experience ledger history
exports.getVendorById = async (req, res, next) => {
  try {
    const { id } = req.params;
    const vendor = await Vendor.findById(id);

    if (!vendor) {
      return res.status(404).json({ success: false, error: 'Vendor profile not found' });
    }

    // Explicitly query the ledger history referencing this vendor
    const historyLedger = await Ledger.find({ vendorId: id }).sort({ createdAt: -1 });

    res.status(200).json({
      success: true,
      data: {
        vendor,
        historyLedger
      }
    });
  } catch (error) {
    next(error);
  }
};

// Administrative status override (Approve, Escalate, Review, etc.)
exports.updateVendorStatus = async (req, res, next) => {
  try {
    const { id } = req.params;
    const { status } = req.body;

    const allowedStatuses = ['Active', 'Under Review', 'Escalated', 'Strategic Partner'];
    if (!allowedStatuses.includes(status)) {
      return res.status(400).json({ success: false, error: `Invalid status option. Must be one of: ${allowedStatuses.join(', ')}` });
    }

    const updatedVendor = await Vendor.findByIdAndUpdate(
      id,
      { status },
      { new: true, runValidators: true } // Return updated document and fire schema rules
    );

    if (!updatedVendor) {
      return res.status(404).json({ success: false, error: 'Vendor profile not found' });
    }

    res.status(200).json({ success: true, message: 'Vendor operational status updated successfully', data: updatedVendor });
  } catch (error) {
    next(error);
  }
};
Why I structured it this way: > Regular expressions ($regex with 'i') are implemented on string attributes like city and category to create flexible filtering inside your Discovery Hub. Instead of using a complex Mongoose virtual populate, getVendorById runs an isolated, high-performance query on the Ledger collection sorting by createdAt: -1 to guarantee institutional timelines appear chronologically.

2. backend/src/controllers/ledgerController.js
JavaScript
// Path: backend/src/controllers/ledgerController.js
const Ledger = require('../models/Ledger');
const Vendor = require('../models/Vendor');

// Add an organizational review, warning, or success story to a target vendor
exports.addLedgerEntry = async (req, res, next) => {
  try {
    const { vendorId } = req.params;
    const { author, entryType, message } = req.body;

    // Verify vendor actually exists before creating dangling ledger records
    const vendorExists = await Vendor.exists({ _id: vendorId });
    if (!vendorExists) {
      return res.status(404).json({ success: false, error: 'Cannot append ledger data. Targeted Vendor does not exist.' });
    }

    const newEntry = await Ledger.create({
      vendorId,
      author,
      entryType,
      message
    });

    res.status(201).json({ success: true, message: 'Institutional knowledge ledger entry logged successfully', data: newEntry });
  } catch (error) {
    next(error);
  }
};
Why I structured it this way: > Uses Mongoose’s lightweight Vendor.exists() check instead of a full findById() query. This avoids parsing the entire vendor payload when all you need to do is confirm their existence before anchoring a new ledger entry.

// Path: src/context/VendorContext.jsx
import React, { createContext, useState, useEffect, useContext } from 'react';
import axios from 'axios';

const VendorContext = createContext();

export const VendorProvider = ({ children }) => {
  const [vendors, setVendors] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [currentUserRole, setCurrentUserRole] = useState('Employee'); // Default operational security tier

  const API_BASE_URL = 'http://localhost:5000/api/vendors';

  // Centralized network fetcher
  const fetchVendors = async (filters = {}) => {
    setLoading(true);
    try {
      // Formulate dynamic URL strings using URLSearchParams from active frontend dashboards
      const queryParams = new URLSearchParams(filters).toString();
      const response = await axios.get(`${API_BASE_URL}?${queryParams}`);
      if (response.data.success) {
        setVendors(response.data.data);
        setError(null);
      }
    } catch (err) {
      console.error("API Fetch Error:", err);
      setError(err.response?.data?.error || 'Failed to sync with Vendor Intelligence API Engine');
    } finally {
      setLoading(false);
    }
  };

  // Switch between Employee and Admin views to test authorization layers on the fly
  const toggleUserRole = () => {
    setCurrentUserRole((prevRole) => (prevRole === 'Employee' ? 'Admin' : 'Employee'));
  };

  useEffect(() => {
    fetchVendors();
  }, []);

  return (
    <VendorContext.Provider value={{ vendors, loading, error, currentUserRole, toggleUserRole, refreshVendors: fetchVendors }}>
      {children}
    </VendorContext.Provider>
  );
};

export const useVendors = () => {
  const context = useContext(VendorContext);
  if (!context) {
    throw new Error('useVendors must be explicitly consumed inside an enclosed VendorProvider wrapper');
  }
  return context;
};
Why I structured it this way: > Consolidates all state synchronization logic into a single source of truth. The fetchVendors function handles dynamic, parametrized object filters cleanly. It also exposes a quick toggleUserRole toggle, giving you an elegant way to simulate role-based authorization rules directly in the interface without needing complex JWT structures early on.

// Path: src/components/admin/ApprovalQueue.jsx
import React, { useState } from 'react';
import axios from 'axios';
import { useVendors } from '../../context/VendorContext';
import { ClipboardList, Check, AlertTriangle } from 'lucide-react';

const ApprovalQueue = () => {
  const { vendors, refreshVendors } = useVendors();
  const [actioningId, setActioningId] = useState(null);

  // Isolate operational items flagged under institutional vetting
  const pendingQueue = vendors.filter(v => v.status === 'Under Review');

  const handleStatusChange = async (id, targetStatus) => {
    setActioningId(id);
    try {
      const res = await axios.patch(`http://localhost:5000/api/vendors/${id}/status`, {
        status: targetStatus
      });
      if (res.data.success) {
        refreshVendors(); // Trigger state alignment across all interface blocks
      }
    } catch (err) {
      console.error("Governance transition processing fault:", err);
    } finally {
      setActioningId(null);
    }
  };

  return (
    <div className="bg-white border border-slate-200 rounded-xl p-5 shadow-sm space-y-4">
      <h3 className="text-sm font-bold text-slate-800 uppercase tracking-wider border-b pb-2 flex items-center gap-2">
        <ClipboardList className="w-4 h-4 text-slate-400" />
        <span>Governance Verification Queue ({pendingQueue.length})</span>
      </h3>

      {pendingQueue.length === 0 ? (
        <p className="text-xs text-slate-400 italic py-4 text-center">
          No records are currently flagged for operational vetting reviews.
        </p>
      ) : (
        <div className="divide-y divide-slate-100 max-h-60 overflow-y-auto pr-1">
          {pendingQueue.map((vendor) => (
            <div key={vendor._id} className="py-3.5 flex flex-col sm:flex-row sm:items-center sm:justify-between gap-3 first:pt-0 last:pb-0">
              <div>
                <h4 className="text-xs font-bold text-slate-700">{vendor.name}</h4>
                <p className="text-[10px] font-medium text-slate-400 mt-0.5">
                  {vendor.category} &bull; {vendor.city}
                </p>
              </div>

              <div className="flex items-center gap-2">
                <button
                  disabled={actioningId !== null}
                  onClick={() => handleStatusChange(vendor._id, 'Active')}
                  className="px-2.5 py-1 bg-emerald-50 hover:bg-emerald-100 text-emerald-700 border border-emerald-200 rounded text-[10px] font-bold transition-colors flex items-center gap-1"
                >
                  <Check className="w-3 h-3" />
                  <span>Approve</span>
                </button>
                <button
                  disabled={actioningId !== null}
                  onClick={() => handleStatusChange(vendor._id, 'Escalated')}
                  className="px-2.5 py-1 bg-red-50 hover:bg-red-100 text-red-700 border border-red-200 rounded text-[10px] font-bold transition-colors flex items-center gap-1"
                >
                  <AlertTriangle className="w-3 h-3" />
                  <span>Escalate</span>
                </button>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

import React, { createContext, useState, useContext, useEffect } from 'react';
import axios from 'axios';

const AuthContext = createContext(null);

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(null);
  const [authLoading, setAuthLoading] = useState(true);

  useEffect(() => {
    // Check local synchronization frames to see if user has a persistent operational token cached
    const savedUser = localStorage.getItem('v_intel_user');
    if (savedUser) {
      setUser(JSON.parse(savedUser));
    }
    setAuthLoading(false);
  }, []);

  const login = async (email, password) => {
    try {
      const response = await axios.post('http://localhost:5000/api/auth/login', { email, password });
      
      if (response.data.success) {
        const authUser = response.data.user;
        setUser(authUser);
        localStorage.setItem('v_intel_user', JSON.stringify(authUser));
        localStorage.setItem('v_intel_token', response.data.token);
        return { success: true };
      }
    } catch (error) {
      return {
        success: false,
        error: error.response?.data?.error || 'Authentication server communications blackout.'
      };
    }
  };

  const logout = () => {
    setUser(null);
    localStorage.removeItem('v_intel_user');
    localStorage.removeItem('v_intel_token');
  };

  return (
    <AuthContext.Provider value={{ user, authLoading, login, logout, isAdmin: user?.role === 'Admin' }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be consumed inside an enclosed AuthProvider wrapper configuration frame.');
  }
  return context;
};

import React from 'react';
import { Navigate, useLocation } from 'react-router-dom';
import { useAuth } from '../../context/AuthContext';

const ProtectedRoute = ({ children, allowedRole }) => {
  const { user, authLoading } = useAuth();
  const location = useLocation();

  if (authLoading) {
    return <div className="text-sm font-medium text-slate-400 p-8 animate-pulse">Running verification scans...</div>;

  // Check role authorization layers
  if (allowedRole && user.role !== allowedRole) {
    // If an Employee runs into Admin boundaries, redirect them to the safe discovery view
    return <Navigate to="/" replace />;
  }

  return children;
};
